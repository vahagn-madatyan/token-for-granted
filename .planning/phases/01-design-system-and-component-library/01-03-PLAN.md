1. Create `src/components/ui/GlitchText.tsx`:
   ```tsx
   import { cn } from '~/lib/cn'

   interface GlitchTextProps {
     text: string
     as?: 'h1' | 'h2' | 'h3' | 'h4' | 'span' | 'p'
     className?: string
     active?: boolean  // when true, glitch animates continuously; when false, only on hover
   }

   export function GlitchText({ text, as: Tag = 'span', className, active = false }: GlitchTextProps) {
     return (
       <Tag
         data-text={text}
         className={cn(
           active ? 'glitch' : 'text-glitch-hover',
           'relative cursor-default',
           className
         )}
       >
         {text}
       </Tag>
     )
   }
   ```
   The `data-text` attribute is critical -- glitch.css uses `content: attr(data-text)` for the ::before/::after pseudo-elements that create the color channel shift effect.

2. Create `src/components/ui/NeonButton.tsx` using CVA for variants:
   ```tsx
   import { cva, type VariantProps } from 'class-variance-authority'
   import { cn } from '~/lib/cn'

   const neonButtonVariants = cva(
     'relative font-headline font-black uppercase tracking-[0.2em] overflow-hidden transition-all hover:scale-105 active:scale-95 cursor-crosshair',
     {
       variants: {
         variant: {
           primary: 'bg-accent text-white clipped-bl animate-pulse-glow',
           secondary: 'bg-transparent border border-outline-variant/30 text-secondary hover:border-secondary/50',
           ghost: 'bg-transparent text-primary-container hover:bg-primary-container/10',
         },
         size: {
           sm: 'py-2 px-6 text-xs',
           md: 'py-3 px-8 text-sm',
           lg: 'py-5 px-12 text-xl',
         },
       },
       defaultVariants: {
         variant: 'primary',
         size: 'md',
       },
     }
   )

   interface NeonButtonProps
     extends React.ButtonHTMLAttributes<HTMLButtonElement>,
       VariantProps<typeof neonButtonVariants> {
     children: React.ReactNode
   }

   export function NeonButton({ children, variant, size, className, ...props }: NeonButtonProps) {
     return (
       <button className={cn(neonButtonVariants({ variant, size }), className)} {...props}>
         {/* Hover sweep overlay */}
         <div className="absolute inset-0 translate-x-[-100%] group-hover:translate-x-0 transition-transform duration-300 bg-white/10" />
         <span className="relative z-10 flex items-center gap-4">{children}</span>
       </button>
     )
   }
   ```
   Note: The primary variant uses `bg-accent` (#ff4655 Valorant red), `clipped-bl` for the diagonal cut corner, and `animate-pulse-glow` for the ambient glow effect -- matching the "INITIATE PROTOCOL" button in the Stitch design.

3. Create `src/components/ui/HexCard.tsx`:
   ```tsx
   import { cva, type VariantProps } from 'class-variance-authority'
   import { cn } from '~/lib/cn'

   const hexCardVariants = cva(
     'relative overflow-hidden',
     {
       variants: {
         elevation: {
           low: 'bg-surface-container-low',
           medium: 'bg-surface-container',
           high: 'bg-surface-container-high',
           highest: 'bg-surface-container-highest',
         },
         clip: {
           none: '',
           tr: 'clipped-tr',
           bl: 'clipped-bl',
           full: 'clipped-full',
         },
         glow: {
           none: '',
           cyan: 'shadow-[0_0_30px_rgba(0,245,255,0.15)]',
           accent: 'shadow-[0_0_30px_rgba(255,70,85,0.15)]',
         },
       },
       defaultVariants: {
         elevation: 'low',
         clip: 'tr',
         glow: 'none',
       },
     }
   )

   interface HexCardProps
     extends React.HTMLAttributes<HTMLDivElement>,
       VariantProps<typeof hexCardVariants> {
     borderAccent?: 'primary' | 'secondary' | 'accent' | 'none'
   }

   export function HexCard({
     children,
     elevation,
     clip,
     glow,
     borderAccent = 'primary',
     className,
     ...props
   }: HexCardProps) {
     const borderClasses = {
       primary: 'border-l border-primary-container/30',
       secondary: 'border-l border-secondary/30',
       accent: 'border-l border-accent/30',
       none: '',
     }

     return (
       <div
         className={cn(
           hexCardVariants({ elevation, clip, glow }),
           borderClasses[borderAccent],
           'p-6',
           className
         )}
         {...props}
       >
         {children}
       </div>
     )
   }
   ```
   Key: NO 1px borders for sectioning. Uses tonal surface shifts (surface-container-low vs surface-container-high) and only a subtle left-border accent (border-l with /30 opacity). The `clip` variants add diagonal cut corners per the tactical design spec.
